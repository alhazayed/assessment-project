-- Example: PHQ9 suicide risk detection
UPDATE assessment_submissions
SET high_risk_flag = true
WHERE definition_id = (SELECT id FROM assessment_definitions WHERE code = 'PHQ9')

    SELECT 1 FROM assessment_responses ar
    WHERE ar.submission_id = assessment_submissions.id
    AND ar.item_id IN (SELECT id FROM assessment_items 
                       WHERE definition_id = assessment_definitions.id 
                       AND item_number = 9)
    AND ar.response_value >= 1
  );
